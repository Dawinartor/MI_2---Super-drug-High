How to use Tiles to make a World in Phaser 3:
1. Configure the Preload-Function

    gameScene.preload = function() 
    {
  
    // Map-bezogene JSON Daten werden geladen:
    this.load.tilemapTiledJSON('map', 'Assets/Test02/SpielKarte.json');

    // Lade das TileSet mit spezifischen Namen, den du dem Tileset vergeben hast: 
    ** <NameFürDatei>, <PathDesTileSets>, <GrößeDerTiles> **
    this.load.spritesheet('MarioTiles22', 'Assets/Tiles/Supermario_TileSet.png', {frameWidth : 16, frameHeight : 16});

    }

    gameScene.create = function()
    {
        // Erzeuge eine TileMap durch die, zuvor angelegte tilemapTiledJSON:
        const map = this.make.tilemap( { key : 'map' } );

        // Übergebe Tiles an Variable tiles:
        ** <NameAusspritesheet>, <NameDerPreloadetenDatei>
        const tiles = map.addTilesetImage('MarioTiles','MarioTiles22');

        // Lege Layer fest:
        ** <LayernameInMap>, <VariableDerTiles>, <x>, <y> **
        const backgroundLayer = map.createStaticLayer('Hintergrund', tiles, 0, 0);
        const platformLayer = map.createStaticLayer('Plattform', tiles, 0, 0);
    }
